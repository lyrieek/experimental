2.mapper.xml

<select id="getList" parameterType="java.util.Map" resultMap="BaseResultMap">
  select * from table where 
  <if test="a!= null">
      a = #{a,jdbcType=VARCHAR}
  </if>
  <if test="list!= null">
    and id in
    <foreach item="item" index="index" collection="list" open="(" close=")" separator=",">
     #{item}
    </foreach>
  </if>
</select>
